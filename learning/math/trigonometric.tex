\subsection{Series definitions}


\begin{align*}
  sin(x) &= x - {x^3 \over 3!} + {x^5 \over 5!} - {x^7 \over 7!} + ... =
           \sum_{n=0}^\infty {(-1)^n \over (2n+1)!} x^{2n+1}\\
  cos(x) &= 1 - {x^2 \over 2!} + {x^4 \over 4!} - {x^6 \over 6!} + ... =
           \sum_{n=0}^\infty {(-1)^n \over (2n)!} x^{2n}\\
\end{align*}
